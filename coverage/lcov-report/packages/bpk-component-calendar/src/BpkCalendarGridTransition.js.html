      if (differenceInCalendarMonths(nextProps.month, this.props.month) === 1) {
        // Transition to next month

          isTransitioning: true,

      if (differenceInCalendarMonths(nextProps.month, this.props.month) === -1) {
        // Transition to previous month

          addMonths(nextProps.month, 1),
        ],
      });
